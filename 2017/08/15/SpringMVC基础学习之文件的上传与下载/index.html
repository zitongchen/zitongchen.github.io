SpringMVC基础学习之文件的上传与下载

发表于 2017-08-15 | 分类于 spring | 阅读次数

配置MultipartResolver实现文件上传，SpringMVC专门提供了CommonsMultipartResolver组件用于文件上传：

  • maxUploadSize 文件最大限制，单位是byte
  • maxInMemorySize 低于这个大小的文件暂存内存中
  • defaultEncoding 默认编码

上传的组件部署需要写上id的值，因为DispatcherServlet类定义好了id的属性值为：multipartResolver

上传需要的jar包：

  • commons-fileupload.jar
  • commons-io.jar

public class FileUploadController {
    @RequestMapping(value="/upload/uploadfile", method=RequestMethod.GET)
    public String uploadLogin(){
        return "fileupload";
    }
    @RequestMapping(value="/upload/uploadfile", method=RequestMethod.POST)
    public String uploadFile(@RequestParam("imageFile") MultipartFile imageFile,HttpServletRequest request){
        //获取保存图片的绝对路径（即webapps文件加下的路径）
        String uploadUrl = request.getServletContext().getRealPath("/") + "images/";
        //获取文件名
        String filename = imageFile.getOriginalFilename();
        //File类是操作目录跟文件的，但不读取文件内容
        File dir = new File(uploadUrl);
        if(!dir.exists()){
            dir.mkdirs();
        }
        System.out.println("文件上传到：" + uploadUrl + filename);
        //通过路径获取文件信息，若不存在文件，便创建一个
        File targetFile = new File(uploadUrl + filename);
        if(!targetFile.exists()){
            try{
                targetFile.createNewFile();
            }catch(IOException e){
                e.printStackTrace(); 
            }
        }
        try {
            //把文件复制到指定文件内
            imageFile.transferTo(targetFile);
        } catch (IllegalStateException | IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
      //返回文件路径
        request.setAttribute("imgUrl", request.getContextPath() +"/images/" + filename);
        return "showimg";
    }
  //多文件上传
    @RequestMapping(value="/upload/moreUpload",method = RequestMethod.POST)
    public String moreUpload(HttpServletRequest request) throws IOException{
        //把request转换为MultipartHttpServletRequest
        MultipartHttpServletRequest multipartHttpServletRequest = (MultipartHttpServletRequest) request;
        //获取包含多个文件的map集合
        Map<String, MultipartFile> files = multipartHttpServletRequest.getFileMap();
        //获取保存路径的绝对路径
        String uploadUrl = request.getServletContext().getRealPath("/") + "images/";
        File dir = new File(uploadUrl);
        if(!dir.exists()){
            dir.mkdirs();
        }
        List<String> fileList = new ArrayList<String>();
        //获取FileMap的值，其类型为MultipartFile
        for(MultipartFile file:files.values()){
            File targetFile = new File(uploadUrl + file.getOriginalFilename());
            if(!targetFile.exists()){
                targetFile.createNewFile();
            }
            file.transferTo(targetFile);
            fileList.add(request.getContextPath() + "/images/" + file.getOriginalFilename());
        }
      //返回文件路径
        request.setAttribute("imgList",fileList);
        return "showimg";
    }
}
